cmake_minimum_required(VERSION 3.8)
project(rover_interfaces)

if(CMAKE_COMPILER_IS_GNUCXX OR CMAKE_CXX_COMPILER_ID MATCHES "Clang")
  add_compile_options(-Wall -Wextra -Wpedantic)
endif()

find_package(ament_cmake REQUIRED)
find_package(rosidl_default_generators REQUIRED)
find_package(std_msgs REQUIRED)
find_package(geometry_msgs REQUIRED)
find_package(builtin_interfaces REQUIRED)

# Generate messages
rosidl_generate_interfaces(${PROJECT_NAME}
  # VEX Communication
  "msg/VexStatus.msg"
  "msg/TankDriveCmd.msg"
  "msg/WheelOdom.msg"
  
  # Classification & Landmarks
  "msg/Classification.msg"
  "msg/ClassificationArray.msg"
  "msg/Detection.msg"
  "msg/DetectionArray.msg"
  "msg/Landmark.msg"
  "msg/LandmarkArray.msg"
  "msg/LandmarkVerification.msg"
  
  # Mission & Health
  "msg/MissionHealth.msg"
  "msg/MissionEvent.msg"
  "msg/WaypointList.msg"
  "msg/Waypoint.msg"
  
  # Camera Control
  "msg/CameraParams.msg"
  "msg/ExposureState.msg"
  
  # Bluetooth
  "msg/BeaconSignal.msg"
  "msg/BeaconArray.msg"
  
  DEPENDENCIES std_msgs geometry_msgs builtin_interfaces
)

ament_package()
