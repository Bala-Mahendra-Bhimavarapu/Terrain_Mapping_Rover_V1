# =============================================================================
# Beacon Homing State Message
# =============================================================================
# Reports the complete state of the autonomous beacon homing system
#
# States:
#   SEARCHING   - Spinning/exploring to find beacon signal
#   APPROACHING - Moving toward beacon (signal detected)
#   AVOIDING    - Obstacle detected, maneuvering around it
#   ARRIVED     - Within arrival distance of beacon
#   LOST        - Beacon signal lost for too long
# =============================================================================

std_msgs/Header header

# ===== SYSTEM STATE =====
bool enabled                        # Is homing enabled?
string state                        # Current state (SEARCHING, APPROACHING, etc.)

# ===== BEACON TRACKING =====
bool beacon_detected                # Is beacon currently detected?
float32 beacon_distance             # Estimated distance to beacon (meters, -1 if unknown)
int32 beacon_rssi                   # Signal strength (dBm, 0 if not detected)
bool signal_improving               # Is signal strength improving? 
float32 estimated_direction         # Estimated direction to beacon (radians, robot frame)
float32 confidence                  # Tracking confidence (0.0 - 1.0)
float32 time_since_detection        # Time since last beacon detection (seconds)

# ===== OBSTACLE DETECTION =====
bool obstacle_detected              # Is there an obstacle in front?
float32 obstacle_distance           # Distance to nearest front obstacle (meters)
bool left_blocked                   # Is left side blocked?
bool right_blocked                  # Is right side blocked? 

# ===== CURRENT COMMAND =====
float32 linear_velocity             # Current linear velocity command (m/s)
float32 angular_velocity            # Current angular velocity command (rad/s)

# ===== STATUS MESSAGE =====
string message                      # Human-readable status message

# ===== STATISTICS =====
uint32 total_readings               # Total beacon readings received
uint32 successful_approaches        # Times signal improved while approaching
float32 distance_traveled           # Total distance traveled (meters)
float32 elapsed_time                # Time since homing started (seconds)
